Quick start

Define your client/server API using vanilla JS classes. 

class ClientClass {
                        async "admin:register"(clientId) {
                            this.id = clientId;
                        }

                        async "users:joined"(userId) {
                            console.log(`${this.id}> ${userId} joined`);
                        }
                    }
 
class ServerClass {
                        constructor() {
                            this.users = [];
                        }

                        async onConnect(client) {
                            user = { client, id: uuid() };
                            await client.admin.register(user.id);
                            this.users.push(user);

                            this.users.forEach(
                                ur => u.client.users.joined(user.id)
                            );
                        }
                    }

Load socketless with the client and server class, and use it to build your clients and server. 

const { generateClientServer } = require('socketless');
                    const ClientServer = generateClientServer(ClientClass, ServerClass);

                    ClientServer.createServer().listen(1234, () => {
                        ClientServer.createClient(`http://localhost:1234`);
                        ClientServer.createClient(`http://localhost:1234`);
                        ClientServer.createClient(`http://localhost:1234`);
                    });

And we're done.
